#### ysp9714

```python
def new_board(board):
    return [[num for num in row if num != 0] for row in zip(*board)]

def solution(board, moves):
    answer = 0
    board = new_board(board)
    out = []
    for i in moves:
        i -= 1
        if len(board[i]):
            popped = board[i].pop(0)
            if out and out[-1] == popped:
                out.pop()
                answer += 2
            else:
                out.append(popped)
    return answer
```

```javascript
function solution(board, moves) {
  let cnt = 0;
  const stack = [];
  const size = Math.max.apply(null, moves);

  for (const move of moves) {
    for (let i = 0; i < size; i++) {
      const target = board[i][move - 1];
      if (target) {
        board[i][move - 1] = 0;
        if (stack.length > 0 && stack[stack.length - 1] === target) {
          // 연이든 값이 같은 인형인 경우
          cnt += 2;
          stack.pop();
        } else {
          // 인형 쌓기
          stack.push(target);
        }
        break;
      }
    }
  }
  return cnt;
}
```
