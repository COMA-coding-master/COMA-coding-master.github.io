```python
def solution(n):
    chk = n
    ans = []
    while chk:
        ans.append(chk % 3)
        chk //= 3
    ans.reverse()
    answer = 0
    for i, v in enumerate(ans):
        answer += v * (3**i)
    return answer
```

```python
def reverse(n: int):
    t = []
    while True:
        if n < 3:
            t.append(n)
            return t
        x = n % 3
        t.append(x)
        n = n // 3

def three2ten(n: list):
    start_num = 1
    result = 0
    while n:
        x = n.pop()
        result += x * start_num
        start_num *= 3
    return result

def solution(n):
    return three2ten(reverse(n))
```

```javascript
function solution(n) {
  let quot = n;
  let result = "";

  while (quot > 0) {
    result += quot % 3;
    quot = parseInt(quot / 3);
  }
  return parseInt(result, 3);
}
```

```java
class Solution {
  public int solution(int n) {

    while(n > 0){
      str += (n % 3);
      n /= 3;
    }

    answer = Integer.parseInt(str, 3);
    return answer;
  }
```
