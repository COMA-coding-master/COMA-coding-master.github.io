``` java
class Solution {
    public long solution(int price, int money, int count) {
        
        long answer = -1;
        long sum = 0;

        for(int i = 1; i <= count; i++){
                sum += price * i;
        }

        answer = sum - money;

        // 금액이 부족하지 않을 경우 
        if(answer < 0){
            answer = 0;
        }

        return answer;
    }
}
```

``` java
public long solution(int price, int money, int count) {
    long totalMoney = money * -1;

    for (int i = 1; i <= count; i++) {
        totalMoney += price * i;
    }

    if(totalMoney <= 0) {
        return 0;
    }

    return totalMoney;
}
```
