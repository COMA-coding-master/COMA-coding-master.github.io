```javascript
function solution(numbers) {
  let sum = 0;
  const sortedNumbers = numbers.sort((a, b) => a - b);

  for (let i = 0; i < 10; i++) {
    if (sortedNumbers.length === 0 || sortedNumbers[0] !== i) {
      sum += i;
    } else {
      sortedNumbers.shift();
    }
  }
  return sum;
}
```

class Solution {
    public int solution(int n) {
        int answer = 0;
        
        for(int i = 2; i < n; i++) {
            if(n % i == 1) {
                answer = i;
                break;
            }
        }
        return answer;
    }
}
