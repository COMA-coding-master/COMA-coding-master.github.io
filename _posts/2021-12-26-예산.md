#### sjw96007

``` python
def solution(d, budget):
    a = d; chk = budget; flag=True
    a.sort()
    for idx, w in enumerate(a):
        chk -= w
        if chk < 0:
            break
    if chk >= 0:
        flag = False
    answer = idx if flag else idx+1
    return answer
```

```python
def solution(d, budget):
    d.sort()
    total, answer = 0, 0
    for i in d:
        if total + i <= budget:
            total += i
            answer += 1
        else:
            return answer
    return answer
```

```javascript
function solution(d, budget) {
  let start = 0, end = d.length;
  let targetArray;

  d = d.sort((a, b) => a - b); // 작은 예산 순으로 정렬

  while (end > 0) {
    targetArray = d.slice(start, end)
    if (targetArray.reduce((stack, el)=>(stack+el), 0) <= budget) return targetArray.length;
    end -= 1
  }
  return 0;
}
```

public int solution(int[] d, int budget) {
    int answer = 0;
    Arrays.sort(d);

    for (int i = 0; i < d.length; i++) {
        if(budget >= d[i]){
            budget -= d[i];
            answer = i+1;
        } else {
            break;
        }
    }
    return answer;
}
