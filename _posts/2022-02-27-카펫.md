class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = {};
        int area = 0;
        int row = 0;
        int col = 0;
        
        answer = new int[2];
        area = brown + yellow;
 
        for (int i = 1; i <= area; i++) {
            row = i;
            col = area / row;
            
            if (row > col) {
                continue;
            }

            if ((row - 2) * (col - 2) == yellow) {
                answer[0] = col;
                answer[1] = row;
                return answer;
            }
        }

        return answer;
    }
}
```

function solution(brown, yellow) {
  let total = brown + yellow;
  let init = 3;

  while (true) {
    if (init >= total / init && (init - 2) * (total / init - 2) === yellow) {
      break;
    }
    init += 1;
  }

  return [init, total / init];
}
```
