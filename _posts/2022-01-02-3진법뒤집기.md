```python
def solution(n):
    invert_num = ''

    while n > 0 :
        n, mod = divmod(n, 3)
        invert_num += str(mod)

    return int(str(invert_num), base = 3)
```

```python
def solution(n):
    reverse = ''
    while n:
        reverse += str(n % 3)
        n = n // 3

    return int(reverse, 3)
```

```javascript
function solution(n) {
  let quot = n;
  let result = "";

  while (quot > 0) {
    result += quot % 3;
    quot = parseInt(quot / 3);
  }
  return parseInt(result, 3);
}
```

```java
class Solution {
  public int solution(int n) {
    int answer = 0;
    String str = "";

    while(n > 0){
      str += (n % 3);
      n /= 3;
    }

    answer = Integer.parseInt(str, 3);
    return answer;
  }
}
```
