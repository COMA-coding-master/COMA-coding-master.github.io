```python
def solution(n):
    invert_num = ''

    while n > 0 :
        n, mod = divmod(n, 3)
        invert_num += str(mod)

    return int(str(invert_num), base = 3)
```

```javascript
function solution(n) {
  let quot = n;
  let result = "";

  while (quot > 0) {
    result += quot % 3;
    quot = parseInt(quot / 3);
  }
  return parseInt(result, 3);
}
```
